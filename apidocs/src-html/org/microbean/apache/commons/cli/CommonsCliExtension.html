044/**
045 * A <a
046 * href="http://docs.jboss.org/cdi/spec/2.0-PRD/cdi-spec.html#spi"
047 * target="_parent">CDI 2.0 portable extension</a> that exposes the <a
048 * href="https://commons.apache.org/proper/commons-cli/"
049 * target="_parent">Apache Commons CLI project</a> as CDI beans.
050 *
051 * @author <a href="https://about.me/lairdnelson"
052 * target="_parent">Laird Nelson</a>
053 *
054 * @see CommandLine
055 *
056 * @see CommandLineParser
057 *
058 * @see Option
059 *
060 * @see Options
061 */
062public class CommonsCliExtension implements Extension {
063
064
065  /*
066   * Constructors.
067   */
068
069  
070  /**
071   * Creates a new {@link CommonsCliExtension}.
072   */
073  public CommonsCliExtension() {
074    super();
075  }
076
077
078  /*
079   * Instance methods.
080   */
081  
082
083  /**
084   * Adds the {@link DefaultParser} and {@link HelpFormatter} classes
085   * as annotated types in {@linkplain ApplicationScoped application
086   * scope} and {@link Singleton} scope respectively.
087   *
088   * @param event the {@link BeforeBeanDiscovery} event indicating
089   * that bean discovery has started; may be {@code null} in which
090   * case no action will be taken
091   *
092   * @see BeforeBeanDiscovery#addAnnotatedType(Class, String)
093   *
094   * @see DefaultParser
095   *
096   * @see HelpFormatter
097   */
098  private final void beforeBeanDiscovery(@Observes final BeforeBeanDiscovery event) {
099    if (event != null) {
100      event.addAnnotatedType(DefaultParser.class, "commons-cli").add(ApplicationScoped.Literal.INSTANCE);
101      event.addAnnotatedType(HelpFormatter.class, "commons-cli").add(SingletonLiteral.INSTANCE);
102      // Somehow, the Producers nested class below is automatically
103      // discovered.  I think this is a bug.  If it is fixed, then
104      // uncomment this line.
105      // event.addAnnotatedType(Producers.class, "commons-cli").add(ApplicationScoped.Literal.INSTANCE);
106    }
107  }
108
109
110  /*
111   * Inner and nested classes.
112   */
113
114
115  /**
116   * A class housing several <a
117   * href="http://docs.jboss.org/cdi/spec/2.0-PRD/cdi-spec.html#producer_method"
118   * target="_parent">producer methods</a> that produce certain <a
119   * href="https://commons.apache.org/proper/commons-cli/"
120   * target="_parent">Apache Commons CLI components</a>.
121   *
122   * @author <a href="https://about.me/lairdnelson"
123   * target="_parent">Laird Nelson</a>
124   */
125  private static final class Producers {
126
127
128    /*
129     * Constructors.
130     */
131
132
133    /**
134     * Creates a new {@link Producers} instance.
135     */
136    private Producers() {
137      super();
138    }
139    
140
141    /*
142     * Static methods.
143     */
144
145
146    /**
147     * {@linkplain Produces Produces} a {@link CommandLine} object in
148     * {@linkplain ApplicationScoped application} scope.
149     *
150     * <p>This method never returns {@code null}.</p>
151     *
152     * @param parser a {@link CommandLineParser}; must not be {@code
153     * null}
154     *
155     * @param optionsInstance an {@link Instance} of {@link Options};
156     * may be {@code null} or {@linkplain Instance#isUnsatisfied()
157     * unsatisfied} in which case a new, "empty" {@link Options}
158     * instance will be parsed instead
159     *
160     * @param commandLineArgumentsInstance an {@link Instance} representing
161     * the actual command line arguments; may be {@code null} or
162     * {@linkplain Instance#isUnsatisfied()}
163     *
164     * @return a {@link CommandLine} instance; never {@code null}
165     *
166     * @exception NullPointerException if {@code parser} is {@code null}
167     *
168     * @exception ParseException if there was a problem parsing the
169     * command line arguments
170     *
171     * @see CommandLineParser#parse(Options, String[])
172     */
173    @Produces
174    @ApplicationScoped
175    private static final CommandLine produceCommandLine(final CommandLineParser parser,
176                                                        final Instance<Options> optionsInstance,
177                                                        @Named("commandLineArguments") final Instance<String[]> commandLineArgumentsInstance)
178      throws ParseException {      
179      Objects.requireNonNull(parser);      
180      final Options options;
181      if (optionsInstance != null && !optionsInstance.isUnsatisfied()) {
182        options = optionsInstance.get();
183      } else {
184        options = new Options();
185      }
186      assert options != null;
187      final String[] commandLineArguments;
188      if (commandLineArgumentsInstance != null && !commandLineArgumentsInstance.isUnsatisfied()) {
189        commandLineArguments = commandLineArgumentsInstance.get();
190      } else {
191        commandLineArguments = null;
192      }
193      final CommandLine returnValue = parser.parse(options, commandLineArguments, true);
194      return returnValue;
195    }
196    
197  }
